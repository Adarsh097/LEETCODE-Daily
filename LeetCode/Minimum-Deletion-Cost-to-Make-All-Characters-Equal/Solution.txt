1class Solution {
2public:
3    long long minCost(string s, vector<int>& c) {
4        long long t = 0, mx = 0;
5        int n = s.size();
6        vector<long long> v(26, 0);
7        for (int x : c) t += x;
8        for (int i = 0; i < n; ++i) v[s[i] - 'a'] += c[i];
9        for (long long x : v) mx = max(mx, x);
10        return t - mx;
11    }
12};
